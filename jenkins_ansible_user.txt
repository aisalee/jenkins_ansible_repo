#Create the user 
useradd ansible 

##Create the password
echo "1234" | passwd --stdin ansible

##Switch as ansible user
su - ansible

#Create the home directory
mkdir /home/ansible/.ssh
chmod 700 /home/ansible/.ssh
cd /home/ansible/.ssh
ssh-keygen
Copy the .pub to the authorized_keys file in the ansible client nodes.

logout

#Enable sudo privileges for the user
vi /etc/sudoers.d/ansible
ansible ALL=(ALL) NOPASSWD:ALL
